let aiHistory = [];

async function sendAIMsg() {
  const inp = document.getElementById('aiInput');
  const msg = inp.value.trim();
  if (!msg) return;

  addAIMsg(msg, 'user');
  inp.value = '';
  document.getElementById('aiSugs').style.display = 'none';

  // Adiciona ao histórico
  aiHistory.push({ role: 'user', content: msg });

  // Mostra indicador de digitando
  const thinkId = showAIThinking();

  try {
    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 400,
        system: BG_SYSTEM_PROMPT,
        messages: aiHistory
      })
    });

    removeAIThinking(thinkId);

    if (!response.ok) throw new Error('API error ' + response.status);

    const data = await response.json();
    const reply = data.content?.[0]?.text || 'Desculpe, tente novamente!';

    aiHistory.push({ role: 'assistant', content: reply });
    // Limita histórico a 10 mensagens para não explodir o contexto
    if (aiHistory.length > 10) aiHistory = aiHistory.slice(-10);

    addAIMsg(reply, 'bot');

  } catch (err) {
    removeAIThinking(thinkId);
    // Fallback para respostas locais se API falhar
    aiHistory.pop(); // remove mensagem sem resposta
    const fallback = getFallbackResponse(msg);
    addAIMsg(fallback, 'bot');
  }
}

// Fallback local caso a API falhe (mantém experiência)
function getFallbackResponse(msg) {
  const m = msg.toLowerCase();
  if (m.includes('plano') || m.includes('preço') || m.includes('valor'))
    return '📋 Temos 3 planos: Review 45min (R$29,90), Review Pro (R$55,90) e BG Elite (R$189,90). Para detalhes completos, veja a seção de Planos no site ou entre no Discord!';
  if (m.includes('pagamento') || m.includes('pix'))
    return '💳 O pagamento é via PIX! Fale com o BGFTN no Discord, informe o plano desejado e se é SOLO ou DUO. Após o PIX, envie o comprovante para liberação em até 24h.';
  if (m.includes('discord') || m.includes('servidor'))
    return '🎮 Acesse: discord.gg/CfRMETMYF — Após entrar, mande DM para o BGFTN com seu comprovante.';
  return '🤖 No momento estou com dificuldade de conexão. Fale diretamente com o BGFTN no Discord para ajuda imediata: discord.gg/CfRMETMYF';
}

function showAIThinking() {
  const id = 'think_' + Date.now();
  const box = document.getElementById('aiMsgs');
  const div = document.createElement('div');
  div.id = id;
  div.className = 'ai-thinking';

// ============ CATEGORIAS ============
function openCategoryModal(id) {
  document.getElementById('categoryModalTitle').textContent = id ? 'Editar Categoria' : 'Nova Categoria';
  document.getElementById('catEditId').value = id || '';
  if (id) {
    const cat = DB.categories().find(c => c.id === id);
    if (cat) { document.getElementById('catName').value = cat.name; document.getElementById('catSlug').value = cat.slug; }
  } else {
    document.getElementById('catName').value = '';
    document.getElementById('catSlug').value = '';
  }
  document.getElementById('categoryModal').classList.add('open');
}
function saveCategory() {
  const id = document.getElementById('catEditId').value;
  const name = document.getElementById('catName').value.trim();
  const slug = document.getElementById('catSlug').value.trim().toLowerCase().replace(/\s+/g, '-');
  if (!name) { toast('Informe o nome da categoria', 'error'); return; }
  const cats = DB.categories();
  const cat = { id: id ? parseInt(id) : Date.now(), name, slug };
  if (id) { const i = cats.findIndex(c => c.id === parseInt(id)); if (i > -1) cats[i] = cat; else cats.push(cat); }
  else cats.push(cat);
  DB.saveCategories(cats);
  closeModal('categoryModal');
  renderAdminCategories();
  toast('Categoria salva!', 'success');
}
function deleteCategory(id) {
  if (!confirm('Remover categoria?')) return;
  DB.saveCategories(DB.categories().filter(c => c.id !== id));
  renderAdminCategories();
  toast('Categoria removida!', 'success');
}
function renderAdminCategories() {
  const cats = DB.categories();
  const prods = DB.products();
  const plans = DB.plans();
  document.getElementById('adminCategoriesTable').innerHTML = cats.map(c => `
    <tr>
      <td style="color:#fff;font-weight:600">${c.name}</td>
      <td style="font-family:monospace;color:var(--muted)">${c.slug}</td>
      <td>${prods.filter(p => p.category === c.slug).length + (c.slug === 'planos' ? plans.length : 0)}</td>
      <td>
        <button class="action-btn action-edit" onclick="openCategoryModal(${c.id})"><i class="ti ti-edit"></i> Editar</button>
        <button class="action-btn action-del" onclick="deleteCategory(${c.id})"><i class="ti ti-trash"></i></button>
      </td>
    </tr>`).join('');
}

function openDigitalModal(id) {
  document.getElementById('digitalModalTitle').textContent = id ? 'Editar Produto' : 'Novo Produto Digital';
  document.getElementById('digEditId').value = id || '';
  const select = document.getElementById('digCategory');

    const prod = DB.products().find(p => p.id === id);
    if (prod) {
      document.getElementById('digName').value = prod.name;
      document.getElementById('digPrice').value = prod.price;
      document.getElementById('digCategory').value = prod.category;
      document.getElementById('digDesc').value = prod.desc;

    ['digName','digPrice','digDesc','digFile'].forEach(i => document.getElementById(i).value = '');

function saveDigital() {
  const id = document.getElementById('digEditId').value;
  const prod = {
    id: id ? parseInt(id) : Date.now(),
    name: document.getElementById('digName').value.trim(),
    price: document.getElementById('digPrice').value,
    category: document.getElementById('digCategory').value,
    desc: document.getElementById('digDesc').value,
    file: document.getElementById('digFile').value,
    status: document.getElementById('digStatus').value
  };
  if (!prod.name || !prod.price) { toast('Preencha nome e preço', 'error'); return; }
  const prods = DB.products();
  if (id) { const i = prods.findIndex(p => p.id === parseInt(id)); if (i > -1) prods[i] = prod; else prods.push(prod); }
  else prods.push(prod);
  DB.saveProducts(prods);
  closeModal('digitalModal');
  renderAdminDigital();
  toast('Produto digital salvo!', 'success');
}

// ============ PIX QR CODE REAL ============
// Gera QR code usando api.qrserver.com (gratuita, sem key)
function generatePixQR(pixKey, amount, receiverName) {
  // Monta o payload PIX (BR Code — padrão EMV)
  const payload = buildPixPayload(pixKey, amount, receiverName || 'BG Beginning Game');
  const encoded = encodeURIComponent(payload);
  const qrUrl = `https://api.qrserver.com/v1/create-qr-code/?size=160x160&data=${encoded}&bgcolor=ffffff&color=000000&margin=8`;

  const img = document.getElementById('pixQrImg');
  if (img) {
    img.src = qrUrl;
    img.alt = 'QR Code PIX — ' + amount;
  }

  // Atualiza exibição do valor
  const amtDisplay = document.getElementById('pixQrAmount');
  const confirmVal = document.getElementById('pixConfirmVal');
  const formatted = 'R$ ' + parseFloat(amount).toFixed(2).replace('.', ',');
  if (amtDisplay) amtDisplay.textContent = formatted;
  if (confirmVal) confirmVal.textContent = formatted;
}

// Constrói o payload PIX no padrão BR Code (simplificado)
function buildPixPayload(key, amount, name) {
  function field(id, value) {
    const len = String(value.length).padStart(2, '0');
    return id + len + value;
  }
  const merchantAccountInfo = field('00', 'BR.GOV.BCB.PIX') + field('01', key);
  const merchantAccountInfoField = field('26', merchantAccountInfo);
  const amountField = field('54', parseFloat(amount).toFixed(2));
  const nameClean = name.substring(0, 25).replace(/[^a-zA-Z0-9 ]/g, '');
  const txId = field('05', '***');
  const additionalData = field('62', txId);

  const payload = '000201'
    + merchantAccountInfoField
    + '5204000053039865802BR'
    + amountField
    + '5913' + nameClean.substring(0, 13).padEnd(13)
    + '6008BRASILIA'
    + additionalData
    + '6304';

  // CRC16-CCITT
  const crc = crc16(payload + '6304');
  return payload + crc;
}

function crc16(str) {
  let crc = 0xFFFF;
  for (let i = 0; i < str.length; i++) {
    crc ^= str.charCodeAt(i) << 8;
    for (let j = 0; j < 8; j++) {
      crc = (crc & 0x8000) ? (crc << 1) ^ 0x1021 : crc << 1;
    }
  }
  return ((crc & 0xFFFF).toString(16).toUpperCase().padStart(4, '0'));
}

function copyPix() {
  const key = DB.siteTexts ? DB.siteTexts().pixKey : 'bgbeginninggame@pix.com';
  navigator.clipboard.writeText(key).then(() => toast('Chave PIX copiada!', 'success'));
}

function confirmPayment() {
  const name    = document.getElementById('co-name').value.trim();
  const email   = document.getElementById('co-email').value.trim();
  const cpf     = document.getElementById('co-cpf').value.trim();

  if (!name)    { toast('Informe seu nome completo', 'error'); return; }

  if (!discord) { toast('Informe seu usuário do Discord', 'error'); return; }

  const orderId = 'BG-' + String(Date.now()).slice(-6);
  const total   = cart.reduce((a, i) => a + parseFloat(i.price), 0);

    userId:   DB.currentUser()?.id,

  // Limpa carrinho

  showPage('success');
}
